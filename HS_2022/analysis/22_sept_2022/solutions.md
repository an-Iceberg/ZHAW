    = 3u + 3 + 7

    = 3u + 10

  - g(u + 1) = 0.5 * (u + 1)^2

    = 0.5 * (u^2 + 2u + 1)

    = 0.5 * u^2 + u + 0.5

  - f(y^2) = 3y^2 + 7

  - g(y^2) = 0.5 * (y^2)^2 // = (y^2) * (y^2) => y^(inner + outer)

    = 0.5 * y^4

  - f((z - 3)^2) = 3((z - 3)^2) + 7

    = 3(z^2 - 6z + 9) + 7

    = 3z^2 - 18z + 27 + 7

    = 3z^2 - 18z + 34

  - g((z - 3)^2) = 0.5 * ((z - 3)^2)^2

    = 0.5 * (z - 3)^4
